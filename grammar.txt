

Prog -> Stmts $

Stmts -> Stmt Stmts
 | 位

Stmt -> ValType id = Val Math
    | print Val

Math -> Factor
    | Expr
    | 位

Expr -> plus Val Factor
    | minus Val Factor
    | 位

Factor -> times Val Factor
    | divide Val Factor
    | Expr
    | 位

Val -> id
    | num
    | - num

ValType -> Int
    | Float
    | Double
